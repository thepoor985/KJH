import seaborn as sns
import matplotlib.pyplot as plt
from scipy.stats import mannwhitneyu

generation1_perf = df[df['Generation'] == 1]['Total']
generation2_perf = df[df['Generation'] == 2]['Total']
generation3_perf = df[df['Generation'] == 3]['Total']
generation4_perf = df[df['Generation'] == 4]['Total']
generation5_perf = df[df['Generation'] == 5]['Total']
generation6_perf = df[df['Generation'] == 6]['Total']

generations = [generation1_perf, generation2_perf, generation3_perf, generation4_perf, generation5_perf, generation6_perf]

p_values = []

for i in range(len(generations)):
    row = []
    for j in range(len(generations)):
        if i == j:
            row.append(1.0)
        else:
            p_value = mannwhitneyu(generations[i], generations[j]).pvalue
            row.append(p_value)
    p_values.append(row)

p_value_matrix = pd.DataFrame(p_values, columns=[f'Gen {i+1}' for i in range(6)], index=[f'Gen {i+1}' for i in range(6)])

plt.figure(figsize=(12, 10))
sns.heatmap(p_value_matrix, annot=True, cmap='coolwarm', vmin=0, vmax=1, fmt=".3f", linewidths=0.5, annot_kws={"size": 16})
plt.title('Mann-Whitney U Test p-values between Generations', fontsize=16)
plt.show()

결론: 4세대 포켓몬은 다른 세대 포켓몬보다 유의미하게 성능이 높은 것으로 생각된다.¶
