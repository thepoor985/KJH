import pandas as pd
import matplotlib.pyplot as plt

video['dislike_ratio'] = video['dislikecnt'] / (video['likecnt'] + video['dislikecnt'])

video_sampled = video['videopk'].unique()[:] 

plt.figure(figsize=(12, 6))
for vid in video_sampled:
    sample_df = video[video['videopk'] == vid]
    plt.plot(sample_df['ct'], sample_df['dislike_ratio'], marker='o', label=f'Video {vid}')

plt.xlabel('Time')
plt.ylabel('Dislike Ratio')
plt.title('Dislike Ratio Over Time for Multiple Videos')
plt.legend()
plt.show()
